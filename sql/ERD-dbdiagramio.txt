Table Track {
  id varchar [pk]
  title varchar
  duration int // seconds
  tempo int //bpm
  album_id varchar
}

Table Album {
  id varchar [pk]
  year int
  name varchar
}

Table Artist {
  id varchar [pk]
  name varchar
}


Table AlbumArtist {
  album_id varchar
  artist_id varchar
}

Table Genre {
  id varchar [pk]
  name varchar
}

Table GenreAlbum {
  genre_id varchar
  album_id varchar
}

Ref: GenreAlbum.album_id > Album.id
Ref: GenreAlbum.genre_id > Genre.id

// each songs belongs to one album
Ref: Track.album_id > Album.id

// each artist can have many albums
// each album can have many artists
Ref: AlbumArtist.artist_id > Artist.id
Ref: AlbumArtist.album_id > Album.id
